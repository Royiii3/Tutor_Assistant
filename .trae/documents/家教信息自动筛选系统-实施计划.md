# 家教信息自动筛选系统 - 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: 使用 `superpowers:subagent-driven-development` 或 `superpowers:executing-plans` 来实施此计划。

**目标：** 构建一个全自动、零成本的家教信息筛选系统，自动监听微信群消息、解析家教信息、计算通勤距离、匹配用户条件，并推送符合条件的消息到微信。

**架构概述：** 采用模块化设计，微信消息通过 ComWeChatRobot 接收，经过正则解析或 DeepSeek AI 兜底提取结构化数据，通过高德 API 计算通勤距离，条件匹配后通过 PushPlus 推送到微信。

**技术栈：** Python 3.10+ / ComWeChatRobot / DeepSeek API / 高德地图 API / PushPlus / pytest

import re

TUTOR_KEYWORDS = [
    r"家教", r"辅导", r"教员", r"教老师", r"时薪",
    r"元/小时", r"/小时", r"元/h", r"/h",
    r"学员", r"大学生", r"数理化", r"语文数学英语",
    r"初中", r"高中", r"初三", r"高一", r"高二", r"高三",
    r"科目", r"年级"
]

class PreFilter:
    def __init__(self):
        self.patterns = [re.compile(kw) for kw in TUTOR_KEYWORDS]

    def is_tutor_message(self, text: str) -> bool:
        match_count = sum(1 for p in self.patterns if p.search(text))
        return match_count >= 2

def is_tutor_message(text: str) -> bool:
    pf = PreFilter()
    return pf.is_tutor_message(text)
```

import re
from parser.tutor_job import TutorJob

class FormatAParser:
    ADDRESS_PATTERN = re.compile(r"【地址】(.+?)(?=【|$)", re.DOTALL)
    SUBJECT_PATTERN = re.compile(r"【科目】(.+?)(?=【|$)", re.DOTALL)
    GRADE_PATTERN = re.compile(r"【学员】(初中[一二三四五六]?|高一|高二|高三|高中)", re.DOTALL)
    TIME_PATTERN = re.compile(r"【时间】(.+?)(?=【|$)", re.DOTALL)
    SALARY_PATTERN = re.compile(r"【薪资】(\d+)(?:-(\d+))?[元]?/小时", re.DOTALL)
    TEACHER_PATTERN = re.compile(r"【教员】(.+?)(?=【|$)", re.DOTALL)

    SUBJECT_MAP = {
        "语文": ["语文"], "数学": ["数学"], "英语": ["英语"],
        "科学": ["科学"], "物理": ["物理"], "化学": ["化学"],
        "生物": ["生物"], "地理": ["地理"], "历史": ["历史"],
        "政治": ["政治"]
    }

    def parse(self, text: str) -> TutorJob | None:
        address = self._extract(ADDRESS_PATTERN, text)
        subject_text = self._extract(SUBJECT_PATTERN, text)
        grade = self._extract(GRADE_PATTERN, text) or self._extract_grade_from_text(text)
        time_req = self._extract(TIME_PATTERN, text)
        salary_match = SALARY_PATTERN.search(text)
        teacher_req = self._extract(TEACHER_PATTERN, text)

        salary = int(salary_match.group(1)) if salary_match else None
        salary_max = int(salary_match.group(2)) if salary_match and salary_match.group(2) else None

        subjects = self._parse_subjects(subject_text or "")

        if not address and not subject_text:
            return None

        return TutorJob(
            id="", raw_text=text, source_group="",
            address=address or "", address_coords=None,
            subjects=subjects, grade=grade or "",
            student_info="", time_requirement=time_req or "",
            teacher_requirement=teacher_req or "",
            salary=salary, salary_max=salary_max,
            commute_time=None, commute_distance=None
        )

    def _extract(self, pattern, text):
        match = pattern.search(text)
        return match.group(1).strip() if match else None

    def _extract_grade_from_text(self, text):
        grade_pattern = re.compile(r"(初中[一二三四五六]?|高一|高二|高三|高中)")
        match = grade_pattern.search(text)
        return match.group(1) if match else None

    def _parse_subjects(self, text):
        subjects = []
        for key, vals in self.SUBJECT_MAP.items():
            if key in text:
                subjects.extend(vals)
        return list(set(subjects))
```

import re
from parser.tutor_job import TutorJob

class FormatBParser:
    STAR_PATTERN = re.compile(r"⭐+([^⭐]+)⭐+")
    GRADE_PATTERN = re.compile(r"(初中[一二三四五六]?|高一|高二|高三|高中)")
    SALARY_PATTERN = re.compile(r"(\d+)-(\d+)/[hh]|(\d+)/[hh]")
    SUBJECT_PATTERN = re.compile(r"(语文|数学|英语|科学|物理|化学|生物|地理|历史|政治)")
    TIME_PATTERN = re.compile(r"(周末|周一|周二|周三|周四|周五|周六)[^。，,]*[次个]")

    def parse(self, text: str) -> TutorJob | None:
        if not self.STAR_PATTERN.search(text):
            return None

        location = self._extract_location(text)
        grade = self._extract_grade(text)
        subjects = self._extract_subjects(text)
        salary, salary_max = self._extract_salary(text)
        time_req = self._extract_time(text)

        return TutorJob(
            id="", raw_text=text, source_group="",
            address=location or "", address_coords=None,
            subjects=subjects, grade=grade or "",
            student_info="", time_requirement=time_req or "",
            teacher_requirement="",
            salary=salary, salary_max=salary_max,
            commute_time=None, commute_distance=None
        )

    def _extract_location(self, text):
        match = self.STAR_PATTERN.search(text)
        return match.group(1).strip() if match else None

    def _extract_grade(self, text):
        match = self.GRADE_PATTERN.search(text)
        return match.group(1) if match else None

    def _extract_subjects(self, text):
        return self.SUBJECT_PATTERN.findall(text)

    def _extract_salary(self, text):
        range_match = re.search(r"(\d+)-(\d+)/[hh]", text)
        if range_match:
            return int(range_match.group(1)), int(range_match.group(2))
        single_match = re.search(r"(\d+)/[hh]", text)
        if single_match:
            return int(single_match.group(1)), None
        return None, None

    def _extract_time(self, text):
        match = self.TIME_PATTERN.search(text)
        return match.group(0) if match else ""
```

import json
import requests
from parser.tutor_job import TutorJob

class AIParser:
    SYSTEM_PROMPT = """你是一个家教信息提取助手。请从消息中提取以下信息，返回JSON格式：
{
    "address": "地址",
    "subjects": ["科目列表"],
    "grade": "年级",
    "salary": 时薪数字,
    "salary_max": 最高时薪(如果有范围),
    "time_requirement": "时间要求"
}
只返回JSON，不要其他内容。"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.api_url = "https://api.deepseek.com/chat/completions"

    def parse(self, text: str) -> TutorJob | None:
        if not self.api_key or self.api_key == "YOUR_DEEPSEEK_KEY":
            return None

        try:
            response = requests.post(
                self.api_url,
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json"
                },
                json={
                    "model": "deepseek-chat",
                    "messages": [
                        {"role": "system", "content": self.SYSTEM_PROMPT},
                        {"role": "user", "content": text}
                    ],
                    "temperature": 0.1
                },
                timeout=30
            )
            response.raise_for_status()
            data = response.json()
            content = data["choices"][0]["message"]["content"]
            info = json.loads(content)
            return TutorJob(
                id="", raw_text=text, source_group="",
                address=info.get("address", ""),
                address_coords=None,
                subjects=info.get("subjects", []),
                grade=info.get("grade", ""),
                student_info="",
                time_requirement=info.get("time_requirement", ""),

import requests
from typing import Optional

class Geocoder:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.api_url = "https://restapi.amap.com/v3/geocode/geo"

    def geocode(self, address: str) -> Optional[tuple[float, float]]:
        if not address:
            return None

        try:
            response = requests.get(
                self.api_url,
                params={"key": self.api_key, "address": address, "city": "杭州"},
                timeout=10
            )
            data = response.json()
            if data.get("status") == "1" and data.get("geocodes"):
                location = data["geocodes"][0]["location"]
                lng, lat = location.split(",")
                return (float(lng), float(lat))
            return None
        except Exception:
            return None

import requests
from typing import Optional

class DistanceCalculator:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.api_url = "https://restapi.amap.com/v3/direction/driving"

    def calculate(
        self,
        origin: Optional[tuple[float, float]],
        destination: Optional[tuple[float, float]]
    ) -> Optional[dict]:
        if not origin or not destination:
            return None

        try:
            response = requests.get(
                self.api_url,
                params={
                    "key": self.api_key,
                    "origin": f"{origin[0]},{origin[1]}",
                    "destination": f"{destination[0]},{destination[1]}"
                },
                timeout=10
            )
            data = response.json()
            if data.get("status") == "1":
                path = data["route"]["paths"][0]
                return {
                    "duration": int(path["duration"]) // 60,
                    "distance": int(path["distance"]) / 1000
                }
            return None
        except Exception:
            return None
```

from parser.tutor_job import TutorJob
from config import UserConfig

class Matcher:
    def __init__(self, config: UserConfig):
        self.config = config

    def match(self, job: TutorJob) -> bool:
        if not self._check_salary(job):
            return False
        if not self._check_subjects(job):
            return False
        if not self._check_grade(job):
            return False
        if not self._check_commute_time(job):
            return False
        return True

    def _check_salary(self, job: TutorJob) -> bool:
        if not job.salary:
            return False
        effective_salary = job.salary_max if job.salary_max else job.salary
        return effective_salary >= self.config.min_salary

    def _check_subjects(self, job: TutorJob) -> bool:
        if not job.subjects:
            return True
        return any(s in self.config.subjects for s in job.subjects)

    def _check_grade(self, job: TutorJob) -> bool:
        if not job.grade:
            return True
        return job.grade in self.config.grades

    def _check_commute_time(self, job: TutorJob) -> bool:
        if job.commute_time is None:
            return True
        return job.commute_time <= self.config.max_commute_time
```

import requests
from parser.tutor_job import TutorJob

class PushPlusPusher:
    API_URL = "https://www.pushplus.plus/send"

    def __init__(self, token: str):
        self.token = token

    def build_message(self, job: TutorJob) -> str:
        salary_text = f"{job.salary}"
        if job.salary_max:
            salary_text = f"{job.salary}-{job.salary_max}"
        salary_text += "元/h"

        commute_text = ""
        if job.commute_time:
            commute_text = f"🚗 通勤约 {job.commute_time} 分钟"
            if job.commute_distance:
                commute_text += f" ({job.commute_distance}km)"

        subjects_text = "·".join(job.subjects) if job.subjects else ""

        return f"""┌──────────────────────────────────────┐
  📬 新家教消息
   来自群聊[{job.source_group}]
  ✅ 匹配：{job.id}
  📚 {subjects_text} · {job.grade}
  📍 {job.address}
  💰 {salary_text}
  {commute_text}
  ⏰ {job.time_requirement}
                                      │
└──────────────────────────────────────┘"""

    def push(self, job: TutorJob) -> bool:
        if not self.token or self.token == "YOUR_PUSHPLUS_TOKEN":
            return False

        try:
            message = self.build_message(job)
            response = requests.post(
                self.API_URL,
                params={"token": self.token},
                json={"title": "📬 新家教消息", "content": message},
                timeout=10
            )
            data = response.json()
            return data.get("code") == 200
        except Exception:
            return False

from typing import Callable, Optional
import time

class WeChatListener:
    def __init__(
        self,
        on_message: Optional[Callable[[dict], None]] = None,
        target_groups: Optional[list[str]] = None
    ):
        self.on_message = on_message
        self.target_groups = target_groups or []
        self._running = False
        self._client = None

    def start(self):
        try:
            from com_wechat_robot import WeChatRobot
            self._client = WeChatRobot()
            self._client.start()
            self._running = True
            self._listen_loop()
        except ImportError:
            print("ComWeChatRobot 未安装，使用模拟模式")
            self._running = True
            self._mock_listen()

    def _listen_loop(self):
        while self._running:
            try:
                messages = self._client.get_messages()
                for msg in messages:
                    if self._should_process(msg):
                        if self.on_message:
                            self.on_message(msg)
            except Exception as e:
                print(f"监听错误: {e}")
                time.sleep(5)

    def _should_process(self, msg: dict) -> bool:
        if not self.target_groups:
            return True
        return msg.get("is_group") and msg.get("name") in self.target_groups

    def _mock_listen(self):
        while self._running:
            time.sleep(1)

    def stop(self):
        self._running = False
        if self._client:
            self._client.stop()

    def is_connected(self) -> bool:
        return self._running
```

import sys
import logging
from config import load_config
from parser.parser import Parser
from geo.geocoder import Geocoder
from geo.distance import DistanceCalculator
from matcher import Matcher
from pusher import PushPlusPusher
from wechat_listener import WeChatListener

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s - %(levelname)s - %(message)s"
)
logger = logging.getLogger(__name__)

class TutorAssistant:
    def __init__(self, config_path: str = "config.json"):
        self.config = load_config(config_path)
        self.parser = Parser(vars(self.config))
        self.geocoder = Geocoder(self.config.amap_key)
        self.distance_calc = DistanceCalculator(self.config.amap_key)
        self.matcher = Matcher(self.config)
        self.pusher = PushPlusPusher(self.config.pushplus_token)
        self.my_coords = None
        self._init_location()

    def _init_location(self):
        logger.info(f"正在获取您地址的坐标: {self.config.my_address}")
        self.my_coords = self.geocoder.geocode(self.config.my_address)
        if self.my_coords:
            logger.info(f"坐标获取成功: {self.my_coords}")
        else:
            logger.warning("无法获取坐标，通勤时间计算将被跳过")

    def process_message(self, msg: dict):
        text = msg.get("text", "")
        group = msg.get("name", "")

        logger.info(f"收到群[{group}]消息: {text[:50]}...")

        job = self.parser.parse(text)
        if not job:
            logger.info("消息不是家教相关，跳过")
            return

        job.source_group = group
        job.id = f"{group}_{msg.get('msg_id', '')}"

        if job.address:
            job.address_coords = self.geocoder.geocode(job.address)
            if job.address_coords and self.my_coords:
                distance_info = self.distance_calc.calculate(
                    self.my_coords, job.address_coords
                )
                if distance_info:
                    job.commute_time = distance_info["duration"]
                    job.commute_distance = distance_info["distance"]

        if self.matcher.match(job):
            logger.info(f"✅ 匹配成功: {job.address} - {job.salary}元/h")
            if self.pusher.push(job):
                logger.info("推送成功")
            else:
                logger.warning("推送失败")
        else:
            logger.info("❌ 不符合条件，跳过")

    def run(self):
        logger.info("家教助手启动...")

        listener = WeChatListener(
            on_message=self.process_message,
            target_groups=self.config.target_groups
        )

        try:
            listener.start()
        except KeyboardInterrupt:
            logger.info("收到退出信号，正在停止...")
            listener.stop()

if __name__ == "__main__":
    config_path = sys.argv[1] if len(sys.argv) > 1 else "config.json"
    app = TutorAssistant(config_path)
    app.run()
